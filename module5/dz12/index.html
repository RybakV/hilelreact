    document.querySelectorAll('a').forEach(function callback(element) {

        href = element.getAttribute('href');
        console.log(element);

        if (href.startsWith('https://')) {
            element.innerHTML += "Link is ok";
        }
        else if (href.startsWith('http://')) {
            element.innerHTML += "Link is not safe";
        }
        else {
            element.innerHTML += "Link had no protocol. Added http.";
            element.setAttribute('href', 'https://www.' + href.split('www.')[1]);
        }

        return element;
    });
